> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship `ollie-fleetctl` — the stable verb CLI that Ollie Fleet (the renamed Plesky dashboard) invokes over SSH to manage a Hermes-stack VPS — plus the heartbeat timer and installer script.

**Architecture:** A single-file Python 3 (stdlib-only) CLI in `templates/bin/`, installed to `~/.local/bin` + `/usr/local/bin` symlink by a new `scripts/10-install-fleetctl.sh` (same pattern as script 09 / `ollie-set-identity`). Every verb prints exactly one JSON object to stdout (exit 0) or `{"error": "..."}` (exit non-zero); long verbs stream NDJSON events; `backup` streams a raw tarball. A systemd user timer runs `ollie-fleetctl heartbeat` every 5 minutes, POSTing health to the Fleet server. Design spec: `D:\devprojects\plesky\docs\superpowers\specs\2026-06-10-ollie-fleet-retarget-design.md`.

**Tech Stack:** Python 3 stdlib (argparse, json, subprocess, urllib), bash installer, systemd user units, stdlib `unittest` for tests (run on the dev box with all host commands mocked — fleetctl targets Linux but tests must pass on Windows).

**Repo:** `D:\devprojects\ollie-hermes-install` (all paths below relative to repo root unless absolute).

**The cross-repo health contract** (Plan 2 — the Fleet server — consumes exactly this shape; do not rename keys):

```json
{
  "fleetctlVersion": "1.0.0",
  "collectedAt": "2026-06-10T18:00:00Z",
  "system": {"cpuPercent": 12, "ramUsedMb": 2048, "ramTotalMb": 7937, "diskUsedGb": 21.4, "diskTotalGb": 78.7},
  "units": [{"unit": "hermes-gateway.service", "active": "active"}],
  "containers": [{"name": "cortex", "state": "running", "status": "Up 3 days"}],
  "versions": {"fleetctl": "1.0.0", "hermes": "hermes 0.4.2"},
  "agents": [{"id": "default", "displayName": "Ollie", "color": "#7c5cff", "provider": "anthropic", "model": "...", "gatewayPort": 8642, "dashboardPort": 9119, "needsIdentity": false}],
  "errors": []
}
```

Collectors degrade independently: a failed collector sets its key to `null` and appends `"<key>: <reason>"` to `errors`. `health` never hard-fails. The `agents` array is whatever the orchestrator's `GET /v1/agents` returns, passed through verbatim.

import importlib.machinery
import importlib.util
import io
import json
import os
import pathlib
import tempfile
import unittest
from contextlib import redirect_stdout

ROOT = pathlib.Path(__file__).resolve().parents[1]
FLEETCTL_PATH = ROOT / "templates" / "bin" / "ollie-fleetctl"


def load_fleetctl():
    """Load the extensionless script as a module (fresh copy per call)."""
    loader = importlib.machinery.SourceFileLoader("fleetctl", str(FLEETCTL_PATH))
    spec = importlib.util.spec_from_loader("fleetctl", loader)
    mod = importlib.util.module_from_spec(spec)
    loader.exec_module(mod)
    return mod


def run_main(mod, argv):
    """Run mod.main(argv); return (exit_code, parsed-stdout-lines)."""
    buf = io.StringIO()
    code = 0
    try:
        with redirect_stdout(buf):
            mod.main(argv)
    except SystemExit as e:
        code = e.code if isinstance(e.code, int) else 1
    lines = [json.loads(l) for l in buf.getvalue().splitlines() if l.strip()]
    return code, lines

#!/usr/bin/env python3
"""ollie-fleetctl — stable management verbs for the Ollie/Hermes stack.

Invoked over SSH by the Ollie Fleet dashboard. Contract: every verb prints
exactly one JSON object to stdout and exits 0, or {"error": "..."} and exits
non-zero. Long verbs (agents create, update) stream NDJSON events ending in
{"event": "done"} or {"event": "error"}. `backup` is the one exception: it
streams a raw tarball to stdout. stderr is human-only; Fleet never parses it.

Installed by scripts/10-install-fleetctl.sh. Versioned with the stack on
purpose: the verb names are the stable contract, the implementation follows
whatever the current install scripts set up.
"""

import argparse
import json
import os
import shutil
import subprocess
import sys
import tempfile
import time
import urllib.error
import urllib.request

VERSION = "1.0.0"

HOME = os.path.expanduser("~")
HERMES_DIR = os.path.join(HOME, ".hermes")
PROFILES_DIR = os.path.join(HERMES_DIR, "profiles")
STACK_DIR = os.environ.get("HERMES_STACK_DIR", os.path.join(HOME, "hermes-stack"))
COMPOSE_FILE = os.path.join(STACK_DIR, "docker-compose.yml")
INSTALL_DIR = os.environ.get("OLLIE_INSTALL_DIR", os.path.join(HOME, "ollie-hermes-install"))
ORCH_DIR = os.path.join(HOME, "ollie-hermes-orchestrator")
ORCH_URL = os.environ.get("ORCH_URL", "http://localhost:9123")
ORCH_ENV = os.environ.get("ORCH_ENV", os.path.join(HOME, ".config", "ollie-orchestrator", ".env"))
FLEET_ENV = os.environ.get("FLEET_ENV", os.path.join(HOME, ".config", "ollie-fleet", ".env"))


# ---------------------------------------------------------------- plumbing

def emit(obj):
    """Print the verb's single JSON result object."""
    print(json.dumps(obj))


def fail(msg, code=1):
    """Print {"error": ...} and exit non-zero."""
    print(json.dumps({"error": msg}))
    sys.exit(code)


def event(**kw):
    """Print one NDJSON progress event (flushed so SSH relays it live)."""
    print(json.dumps(kw), flush=True)


def run_cmd(args, timeout=30, input_text=None):
    """Run a command; return (rc, stdout, stderr). Never raises on non-zero."""
    try:
        p = subprocess.run(args, capture_output=True, text=True,
                           timeout=timeout, input=input_text)
        return p.returncode, p.stdout, p.stderr
    except FileNotFoundError:
        return 127, "", "%s: not found" % args[0]
    except subprocess.TimeoutExpired:
        return 124, "", "%s: timed out after %ss" % (args[0], timeout)


def read_env_file(path):
    """Parse KEY=VALUE lines (quotes stripped); missing file -> {}."""
    env = {}
    try:
        with open(path) as f:
            for line in f:
                line = line.strip()
                if not line or line.startswith("#") or "=" not in line:
                    continue
                k, _, v = line.partition("=")
                env[k.strip()] = v.strip().strip("'\"")
    except OSError:
        pass
    return env


# -------------------------------------------------------------------- main

def main(argv=None):
    p = argparse.ArgumentParser(prog="ollie-fleetctl",
                                description="Management verbs for the Ollie/Hermes stack")
    sub = p.add_subparsers(dest="verb", required=True)
    sub.add_parser("version", help="fleetctl version")

    args = p.parse_args(argv)
    if args.verb == "version":
        emit({"fleetctl": VERSION})


if __name__ == "__main__":
    main()
```

# ------------------------------------------------------- orchestrator client

def orch_key():
    key = read_env_file(ORCH_ENV).get("ORCHESTRATOR_KEY", "")
    if not key:
        raise RuntimeError("no ORCHESTRATOR_KEY in %s" % ORCH_ENV)
    return key


def orch_request(method, path, body=None, timeout=30):
    """JSON request to the local orchestrator -> (status, parsed_or_None, raw)."""
    req = urllib.request.Request(
        ORCH_URL + path, method=method,
        data=json.dumps(body).encode() if body is not None else None,
        headers={"Authorization": "Bearer " + orch_key(),
                 "Content-Type": "application/json"})
    try:
        with urllib.request.urlopen(req, timeout=timeout) as resp:
            raw = resp.read().decode()
            return resp.status, (json.loads(raw) if raw.strip() else None), raw
    except urllib.error.HTTPError as e:
        return e.code, None, e.read().decode(errors="replace")


# ----------------------------------------------------------- health collectors

def _read_file(path):
    with open(path) as f:
        return f.read()


def _proc_stat_cpu():
    """-> (idle_ticks, total_ticks) from the aggregate cpu line."""
    nums = [int(x) for x in _read_file("/proc/stat").splitlines()[0].split()[1:]]
    idle = nums[3] + (nums[4] if len(nums) > 4 else 0)
    return idle, sum(nums)


def get_cpu_percent():
    i1, t1 = _proc_stat_cpu()
    time.sleep(0.25)
    i2, t2 = _proc_stat_cpu()
    dt = t2 - t1
    return round(100 * (1 - (i2 - i1) / dt)) if dt > 0 else 0


def get_memory_mb():
    """-> (used_mb, total_mb) using MemAvailable semantics."""
    info = {}
    for line in _read_file("/proc/meminfo").splitlines():
        k, _, v = line.partition(":")
        if v.strip():
            info[k.strip()] = int(v.strip().split()[0])  # kB
    total = info.get("MemTotal", 0) // 1024
    avail = info.get("MemAvailable", 0) // 1024
    return total - avail, total


def get_system():
    used_mb, total_mb = get_memory_mb()
    du = shutil.disk_usage("/")
    return {
        "cpuPercent": get_cpu_percent(),
        "ramUsedMb": used_mb,
        "ramTotalMb": total_mb,
        "diskUsedGb": round(du.used / 1e9, 1),
        "diskTotalGb": round(du.total / 1e9, 1),
    }


def get_units():
    rc, out, err = run_cmd(["systemctl", "--user", "list-units", "--all",
                            "--no-legend", "--plain",
                            "hermes-*", "ollie-orchestrator*"])
    if rc != 0:
        raise RuntimeError("systemctl failed: %s" % err.strip())
    units = []
    for line in out.splitlines():
        parts = line.split()
        if len(parts) >= 3 and parts[0].endswith(".service"):
            units.append({"unit": parts[0], "active": parts[2]})
    return units


def get_containers():
    rc, out, err = run_cmd(["docker", "compose", "-f", COMPOSE_FILE,
                            "ps", "-a", "--format", "json"])
    if rc != 0:
        raise RuntimeError("docker compose ps failed: %s" % err.strip())
    containers = []
    for line in out.splitlines():
        if not line.strip():
            continue
        c = json.loads(line)
        containers.append({"name": c.get("Service", ""),
                           "state": c.get("State", ""),
                           "status": c.get("Status", "")})
    return containers


def get_versions():
    versions = {"fleetctl": VERSION}
    rc, out, _ = run_cmd(["hermes", "--version"], timeout=15)
    versions["hermes"] = out.strip() if rc == 0 and out.strip() else "unknown"
    return versions


def get_agents_summary():
    status, parsed, raw = orch_request("GET", "/v1/agents")
    if status != 200 or parsed is None:
        raise RuntimeError("orchestrator /v1/agents returned HTTP %s" % status)
    return parsed.get("agents", [])


def build_health():
    health = {"fleetctlVersion": VERSION,
              "collectedAt": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime())}
    errors = []
    g = sys.modules[__name__] if __name__ != "__main__" else sys.modules["__main__"]
    for key, name in (("system", "get_system"), ("units", "get_units"),
                      ("containers", "get_containers"), ("versions", "get_versions"),
                      ("agents", "get_agents_summary")):
        try:
            health[key] = getattr(g, name)()
        except Exception as e:  # collectors degrade independently
            health[key] = None
            errors.append("%s: %s" % (key, e))
    health["errors"] = errors
    return health


def cmd_health(args):
    emit(build_health())
```

def orch_stream(method, path, body, timeout=600):
    """Send a request and relay the orchestrator's SSE stream as NDJSON.
    Returns the last event object (or None if the stream was empty)."""
    req = urllib.request.Request(
        ORCH_URL + path, method=method,
        data=json.dumps(body).encode() if body is not None else None,
        headers={"Authorization": "Bearer " + orch_key(),
                 "Content-Type": "application/json"})
    last = None
    with urllib.request.urlopen(req, timeout=timeout) as resp:
        for line in resp:
            line = line.decode().strip()
            if not line.startswith("data:"):
                continue
            try:
                obj = json.loads(line[5:].strip())
            except ValueError:
                continue
            last = obj
            event(**obj)
    return last
```

def _payload(args):
    """JSON body from --json or stdin."""
    raw = args.json if args.json else sys.stdin.read()
    try:
        return json.loads(raw)
    except ValueError as e:
        fail("invalid JSON payload: %s" % e, 2)


def cmd_agents(args):
    g = sys.modules[__name__] if __name__ != "__main__" else sys.modules["__main__"]
    if args.action == "list":
        status, parsed, raw = g.orch_request("GET", "/v1/agents")
        if status != 200 or parsed is None:
            fail("orchestrator HTTP %s: %s" % (status, raw[:200]))
        emit(parsed)
    elif args.action == "get":
        status, parsed, raw = g.orch_request("GET", "/v1/agents/%s" % args.id)
        if status != 200 or parsed is None:
            fail("orchestrator HTTP %s: %s" % (status, raw[:200]))
        emit(parsed)
    elif args.action == "create":
        last = g.orch_stream("POST", "/v1/agents", _payload(args))
        if not last or last.get("event") != "done":
            sys.exit(1)
    elif args.action == "update":
        status, parsed, raw = g.orch_request("PATCH", "/v1/agents/%s" % args.id, _payload(args))
        if status != 200:
            fail("orchestrator HTTP %s: %s" % (status, raw[:200]))

# 10-install-fleetctl.sh — install ollie-fleetctl (the verb CLI that Ollie Fleet
# invokes over SSH) and the fleet-heartbeat systemd user timer.

# a box that has never been enrolled. Re-run after `hermes update` is NOT

# always safe and picks up new fleetctl versions after `git pull`.

set -euo pipefail

  exit 1
fi

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
SRC_BIN="${SCRIPT_DIR}/../templates/bin/ollie-fleetctl"
SRC_SVC="${SCRIPT_DIR}/../templates/systemd/fleet-heartbeat.service"

DEST_BIN="${HOME}/.local/bin/ollie-fleetctl"
UNIT_DIR="${HOME}/.config/systemd/user"

for f in "${SRC_BIN}" "${SRC_SVC}" "${SRC_TMR}"; do
  [[ -f "${f}" ]] || { echo "error: ${f} not found" >&2; exit 1; }
done

echo "==> installing ollie-fleetctl → ${DEST_BIN} (+ /usr/local/bin symlink)"
mkdir -p "${HOME}/.local/bin"
cp "${SRC_BIN}" "${DEST_BIN}"
chmod 755 "${DEST_BIN}"
sudo ln -sfn "${DEST_BIN}" /usr/local/bin/ollie-fleetctl
echo "    linked /usr/local/bin/ollie-fleetctl"

echo "==> smoke test"
ollie-fleetctl version

echo "==> installing fleet-heartbeat systemd user units"
mkdir -p "${UNIT_DIR}"
cp "${SRC_SVC}" "${UNIT_DIR}/fleet-heartbeat.service"
cp "${SRC_TMR}" "${UNIT_DIR}/fleet-heartbeat.timer"

export XDG_RUNTIME_DIR="${XDG_RUNTIME_DIR:-/run/user/$(id -u)}"
systemctl --user daemon-reload
systemctl --user enable --now fleet-heartbeat.timer
echo "    timer enabled: $(systemctl --user is-active fleet-heartbeat.timer)"

echo
echo "✓ fleetctl installed."
echo "  verbs: health | agents <action> | restart <target> | logs <service> | update | backup | heartbeat | version"
echo "  heartbeat is a no-op until enrollment writes ${HOME}/.config/ollie-fleet/.env"
echo "  (FLEET_URL=... and FLEET_TOKEN=..., mode 600 — the Fleet dashboard does this for you)"
```
